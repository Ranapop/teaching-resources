// read brand name and no of models
void readBrand(char brandName[], int * noOfModels) {
    char line[MAX_LINE];
    gets(line);
    const char delim[2] = ":";
    char * token;
    //brand name
    token = strtok(line, delim);
    strcpy(brandName,token);
    //noOfModels
    token = strtok(NULL,delim);
    *noOfModels = atoi(token);
}


void readModels() {
    char line[MAX_LINE];
    gets(line);
    char * token = strtok(line, "(");
    // while token!=null
    while (token!=NULL) {
        // read model name (until - )
        token = strtok(NULL, "-");
        // read price (until ")" )
        token = strtok(NULL, ")");
        // read space until "("
        token = strtok(NULL, "(");
    }
}
